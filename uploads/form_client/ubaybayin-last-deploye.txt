                let mcScoreAdded = false;

                function checkMCAnswer(button, status) {
                    let buttons = document.querySelectorAll(".mbutton");
                    buttons.forEach(btn => {
                        btn.style.background = "";
                        btn.style.color = "";
                    });

                    let scoreDisplay = document.getElementById("mc-score-display");
                    let resultMessage = document.getElementById("mc-result-message");

                    // Get audio elements
                    let correctSound = document.getElementById("correct-sound");
                    let wrongSound = document.getElementById("wrong-sound");

                    if (status === 'correct') {
                        button.style.background = "#4CAF50"; // Green for correct answer
                        button.style.color = "#fff";
                        resultMessage.textContent = "CORRECT!";
                        resultMessage.style.color = "green";
                        resultMessage.style.fontWeight = "bold";
                        resultMessage.style.fontSize = "24px";
                        resultMessage.style.borderRadius = "8px";

                        resultMessage.style.padding = "5px 10px";

                        correctSound.play(); // Play correct sound

                        if (!mcScoreAdded) {
                            scoreDisplay.style.display = "block";
                            mcScoreAdded = true;
                            sendScore(5);
                        }
                    } else {
                        button.style.background = "#FF5733"; // Red for wrong answer
                        button.style.color = "#fff";
                        resultMessage.textContent = "INCORRECT!";
                        resultMessage.style.color = "red";
                        resultMessage.style.fontWeight = "bold";

                        resultMessage.style.backgroundColor = "white";
                        resultMessage.style.padding = "5px 10px";

                        wrongSound.play(); // Play wrong sound
                    }
                }

                function sendScore(score) {
                    let xhr = new XMLHttpRequest();
                    xhr.open("POST", "", true);
                    xhr.setRequestHeader("Content-Type", "application/x-www-form-urlencoded");
                    xhr.send("score=" + score);

                    xhr.onreadystatechange = function () {
                        if (xhr.readyState == 4 && xhr.status == 200) {
                            console.log("Score saved: " + xhr.responseText);
                        }
                    };
                }

                <script>
                    const correctPray = "RAY";
                    const correctFaith = "AITH";

                    let prayAnswered = false;
                    let faithAnswered = false;

                    function playDing() {
                        const ding = document.getElementById("ding-sound");
                        if (ding) ding.play();
                    }

                    function playWrong() {
                        const wrong = document.getElementById("wrong-sound");
                        if (wrong) wrong.play();
                    }

                    function setInputBorder(ids, color) {
                        ids.forEach(id => {
                            const input = document.getElementById(id);
                            if (input) {
                                input.style.borderColor = color;
                            }
                        });
                    }

                    function autoCheck() {
                        const prayIds = [1, 2, 3].map(i => `pray-${i}`);
                        const faithIds = [1, 2, 3, 4].map(i => `faith-${i}`);

                        const prayVal = prayIds.map(id => document.getElementById(id).value.toUpperCase()).join("");
                        const faithVal = faithIds.map(id => document.getElementById(id).value.toUpperCase()).join("");

                        if (prayVal === correctPray && !prayAnswered) {
                            playDing();
                            setInputBorder(prayIds, "green");
                            prayAnswered = true;
                        } else if (prayVal.length === 3 && prayVal !== correctPray) {
                            playWrong();
                            setInputBorder(prayIds, "red");
                            prayAnswered = false;
                        }

                        if (faithVal === correctFaith && !faithAnswered) {
                            playDing();
                            setInputBorder(faithIds, "green");
                            faithAnswered = true;
                        } else if (faithVal.length === 4 && faithVal !== correctFaith) {
                            playWrong();
                            setInputBorder(faithIds, "red");
                            faithAnswered = false;
                        }
                    }

                    document.querySelectorAll(".letter").forEach(input => {
                        input.addEventListener("input", autoCheck);
                    });
                </script>

                    const correctWisdom = "ISDOM";

                    function playDing() {
                        const ding = document.getElementById("ding-sound");

                        const wrong = document.getElementById("wrong-sound");
                        if (wrong) wrong.play();

                    function autoCheckWords() {
                        const wisdomIds = [2, 3, 4, 5, 6]; // For I-S-D-O-M
                        const learnIds = [1, 2, 3, 4];     // For E-A-R-N

                        let wisdomVal = "";
                        let learnVal = "";

                        wisdomIds.forEach((i, index) => {
                            const el = document.getElementById(`wisdom-${i}`);
                            const correctChar = correctWisdom[index];
                            if (el) {
                                const val = el.value.toUpperCase();
                                wisdomVal += val;
                                if (val === correctChar) {
                                    el.style.borderColor = "green";
                                } else if (val !== "") {
                                    el.style.borderColor = "red";
                                } else {
                                    el.style.borderColor = "#999";
                                }
                            }
                        });

                        learnIds.forEach((i, index) => {
                            const el = document.getElementById(`learn-${i}`);
                            const correctChar = correctLearn[index];
                            if (el) {
                                const val = el.value.toUpperCase();
                                learnVal += val;
                                if (val === correctChar) {
                                    el.style.borderColor = "green";
                                } else if (val !== "") {
                                    el.style.borderColor = "red";
                                } else {
                                    el.style.borderColor = "#999";
                                }
                            }
                        });

                        if (wisdomVal === correctWisdom && !wisdomAnswered) {
                            playDing();
                            wisdomAnswered = true;
                        } else if (wisdomVal.length === 5 && wisdomVal !== correctWisdom) {
                            playWrong();
                            wisdomAnswered = false;
                        }

                        if (learnVal === correctLearn && !learnAnswered) {
                            playDing();
                            learnAnswered = true;
                        } else if (learnVal.length === 4 && learnVal !== correctLearn) {
                            playWrong();
                            learnAnswered = false;
                        }
                    }

                    document.querySelectorAll(".letter").forEach(input => {
                        input.addEventListener("input", autoCheckWords);
                    });

                const images = [
                    'images/f1.png',
                    'images/f2.png',
                    'images/f3.png',

                    'images/f5.png',
                    'images/f6.png',

                ];

                const cardImages = [...images, ...images]; // 8 pairs = 16 cards
                cardImages.sort(() => 0.5 - Math.random()); // Shuffle cards

                const board = document.getElementById('game-board');
                const flipSound = document.getElementById('flip-sound');
                const matchSound = document.getElementById('match-sound');

                let flippedCards = [];
                let lockBoard = false;

                function createCard(image, index) {
                    const card = document.createElement('div');
                    card.classList.add('card');
                    card.dataset.image = image;
                    card.dataset.index = index;

                    card.innerHTML = `
      <div class="card-inner">
        <div class="card-front"></div>
        <div class="card-back">
          <img src="${image}" alt="card image" />
        </div>
      </div>
    `;

                    card.addEventListener('click', () => {
                        if (lockBoard || card.classList.contains('matched') || card === flippedCards[0]) return;

                        // 🔊 Play flip sound
                        flipSound.currentTime = 0;
                        flipSound.play();

                        card.classList.add('flip');
                        flippedCards.push(card);

                        if (flippedCards.length === 2) {
                            lockBoard = true;
                            checkMatch();
                        }
                    });

                    board.appendChild(card);
                }

                function checkMatch() {
                    const [card1, card2] = flippedCards;
                    if (card1.dataset.image === card2.dataset.image) {
                        card1.classList.add('matched');
                        card2.classList.add('matched');

                        // 🔊 Play match sound
                        matchSound.currentTime = 0;
                        matchSound.play();

                        resetFlippedCards();
                    } else {
                        setTimeout(() => {
                            card1.classList.remove('flip');
                            card2.classList.remove('flip');
                            resetFlippedCards();
                        }, 1000);
                    }
                }

                function resetFlippedCards() {
                    flippedCards = [];
                    lockBoard = false;
                }

                // Initialize the board
                cardImages.forEach((img, idx) => createCard(img, idx));

    <script>
        let selectedItems = [];
        let matchedPairs = new Set();
        const totalPairs = document.querySelectorAll("#left-column .item").length;

        function drawLine(start, end) {
            const svg = document.getElementById("line-container");
            const rect = svg.getBoundingClientRect();
            const startRect = start.getBoundingClientRect();
            const endRect = end.getBoundingClientRect();

            const x1 = startRect.left + startRect.width / 2 - rect.left;
            const y1 = startRect.top + startRect.height / 2 - rect.top;
            const x2 = endRect.left + endRect.width / 2 - rect.left;
            const y2 = endRect.top + endRect.height / 2 - rect.top;

            const line = document.createElementNS("http://www.w3.org/2000/svg", "line");
            line.setAttribute("x1", x1);
            line.setAttribute("y1", y1);
            line.setAttribute("x2", x2);
            line.setAttribute("y2", y2);
            line.setAttribute("stroke", "blue");
            line.setAttribute("stroke-width", "4");
            line.setAttribute("stroke-linecap", "round");
            svg.appendChild(line);
        }

        function checkCompletion() {
            if (matchedPairs.size === totalPairs) {
                document.getElementById("congrats-message").style.display = "block";
                document.getElementById("correct-message").style.display = "block";

                // Play ding sound
                const dingSound = new Audio('sounds/ding.mp3');
                dingSound.play();
            }
        }

        document.querySelectorAll(".item").forEach(item => {
            item.addEventListener("click", function () {
                if (selectedItems.length < 2 && !selectedItems.includes(this) && !matchedPairs.has(this.dataset.value)) {
                    this.style.border = "2px solid blue";
                    selectedItems.push(this);
                }

                if (selectedItems.length === 2) {
                    if (selectedItems[0].dataset.value === selectedItems[1].dataset.value) {
                        selectedItems.forEach(el => {
                            el.style.border = "2px solid green";
                            matchedPairs.add(el.dataset.value);
                        });
                        drawLine(selectedItems[0], selectedItems[1]);
                        checkCompletion();
                    } else {
                        setTimeout(() => {
                            selectedItems.forEach(el => el.style.border = "none");
                        }, 500);
                    }
                    selectedItems = [];
                }
            });
        });
    </script>

        document.addEventListener("DOMContentLoaded", () => {
            const answer = ["H", "E", "L", "P"]; // Correct answer
            let hintUsed = false; // Track if the hint is used

            const boxes = document.querySelectorAll(".crossword-box");
            const resultMessage = document.getElementById("result-message");
            const hintButton = document.getElementById("hint-button");

            // Handle hint button click
            hintButton.addEventListener("click", () => {
                if (!hintUsed) {
                    for (let i = 0; i < boxes.length; i++) {
                        if (boxes[i].value.trim() === "") {
                            boxes[i].value = answer[i];
                            boxes[i].readOnly = true; // Prevent editing the revealed hint
                            hintUsed = true;
                            hintButton.disabled = true; // Disable the button
                            hintButton.innerText = "Hint Used!";
                            break;
                        }
                    }
                }
            });

            // Add event listeners to input boxes
            boxes.forEach((box, index) => {
                box.addEventListener("input", () => {
                    // Ensure only uppercase letters are entered
                    box.value = box.value.toUpperCase().trim().charAt(0);
                    checkAnswer();
                });
            });

            // Function to check if all answers are correct
            function checkAnswer() {
                const userAnswer = Array.from(boxes).map(box => box.value.trim().toUpperCase());

                // Ensure all boxes are filled before checking
                if (userAnswer.includes("")) {
                    resultMessage.textContent = ""; // Clear message if not all boxes are filled
                    return;
                }

                if (JSON.stringify(userAnswer) === JSON.stringify(answer)) {
                    resultMessage.textContent = "Correct!";
                    resultMessage.style.color = "green";

                    // Apply glow effect to all boxes
                    boxes.forEach(box => box.classList.add("correct-answer"));
                } else {
                    resultMessage.textContent = ""; // Clear message if incorrect
                    boxes.forEach(box => box.classList.remove("correct-answer"));
                }
            }
        });
    </script>
